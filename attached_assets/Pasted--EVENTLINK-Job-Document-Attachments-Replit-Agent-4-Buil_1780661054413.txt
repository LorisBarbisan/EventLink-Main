Job Document Attachments
Replit Agent 4 Build Instruction — Verified against live codebase


This document provides the complete, codebase-verified instruction for adding file attachments (function sheets, purchase orders) to job postings on EventLink. Employers attach documents. Hired freelancers receive links in their confirmation email and can access documents from their booking card.

Codebase Notes — Read Before Starting
Key findings from codebase review
There is NO separate storage.service.ts. File upload logic lives directly in file.controller.ts and document.controller.ts. Do NOT import from a storage service — follow the same pattern used in file.controller.ts for handling multer uploads and object storage. The server/api/services folder only contains CV parsing and notification scheduler services.

•	server/api/controllers/file.controller.ts — how multer and object storage are used for file uploads

Critical: check the hired status value
The application status value for a hired freelancer may be 'hired', 'accepted', or something else — check the applications table in shared/schema.ts before writing any queries. Use the exact value from the schema, not an assumption.

// ── Upload a document to a job ────────────────────────────────────────────
export async function uploadJobDocument(req: Request, res: Response) {
  try {
    const employerId = req.companyId!;
    const jobId = parseInt(req.params.jobId);
    const { documentType = "other" } = req.body;

    if (isNaN(jobId))
      return res.status(400).json({ error: "Invalid job ID" });

    // Verify job belongs to this employer
    const [job] = await db.select().from(jobs)
      .where(and(eq(jobs.id, jobId), eq(jobs.employerId, employerId)));

    if (!job)
      return res.status(403).json({ error: "Job not found or not yours" });

    if (!req.file)
      return res.status(400).json({ error: "No file provided" });

    if (!ALLOWED_TYPES.includes(req.file.mimetype))
      return res.status(400).json({
        error: "Only PDF, Word, and Excel files are allowed",
      });

    if (req.file.size > MAX_SIZE)
      return res.status(400).json({ error: "File too large. Max 10MB." });

    // ── FILE UPLOAD ───────────────────────────────────────────────────────
    // Use the same upload pattern as file.controller.ts
    // Generate a unique key for storage:
    const fileKey =
      `job-documents/${jobId}/${Date.now()}-${req.file.originalname}`;

    // Replace the lines below with the actual upload call from
    // file.controller.ts, e.g.:
    //   const fileUrl = await uploadToStorage(req.file.buffer, fileKey, ...)
    // ─────────────────────────────────────────────────────────────────────
    const fileUrl = ""; // REPLACE with actual upload call

    const [doc] = await db.insert(jobDocuments).values({
      jobId,
      uploadedByUserId: req.user!.id,

// ── Get documents for a job ───────────────────────────────────────────────
// Employer who owns the job OR a hired freelancer can access
export async function getJobDocuments(req: Request, res: Response) {
  try {
    const userId = req.user!.id;
    const jobId = parseInt(req.params.jobId);

    if (isNaN(jobId))
      return res.status(400).json({ error: "Invalid job ID" });

    const [job] = await db.select().from(jobs).where(eq(jobs.id, jobId));
    if (!job) return res.status(404).json({ error: "Job not found" });

    const isEmployer = req.companyId === job.employerId;

    if (!isEmployer) {
      // ── IMPORTANT: replace 'hired' with the actual hired status
      // value from your applications table schema ──────────────────
      const HIRED_STATUS = "hired"; // CHECK schema — may be 'accepted'
      const [hired] = await db.select().from(applications).where(
        and(
          eq(applications.jobId, jobId),
          eq(applications.freelancerId, userId),
          eq(applications.status, HIRED_STATUS)
        )
      );
      if (!hired)
        return res.status(403).json({
          error: "Documents are only available to hired freelancers",
        });
    }

    const docs = await db.select().from(jobDocuments)
      .where(eq(jobDocuments.jobId, jobId));

    return res.json(docs);
  } catch (err) {
    console.error("getJobDocuments:", err);
    return res.status(500).json({ error: "Failed to fetch documents" });
  }
}

// ── Delete a document (employer only) ────────────────────────────────────
export async function deleteJobDocument(req: Request, res: Response) {
  try {
    const employerId = req.companyId!;
    const docId = parseInt(req.params.docId);

    if (isNaN(docId))
      return res.status(400).json({ error: "Invalid document ID" });

    const result = await db
      .select({ doc: jobDocuments, empId: jobs.employerId })
      .from(jobDocuments)
      .innerJoin(jobs, eq(jobDocuments.jobId, jobs.id))
      .where(eq(jobDocuments.id, docId));

    if (!result.length)
      return res.status(404).json({ error: "Document not found" });

    const { doc, empId } = result[0];

    if (empId !== employerId)
      return res.status(403).json({ error: "Not authorised" });

    // ── Replace with actual delete-from-storage call ──────────────────────
    // Use the same pattern as file.controller.ts for deleting from storage
    // e.g.: await deleteFromStorage(doc.fileKey);
    // ─────────────────────────────────────────────────────────────────────

    await db.delete(jobDocuments).where(eq(jobDocuments.id, docId));
    return res.json({ success: true });
  } catch (err) {
    console.error("deleteJobDocument:", err);
    return res.status(500).json({ error: "Failed to delete document" });

TypeScript — server/api/routes/job-document.route.ts
import { Router } from "express";
import multer from "multer";
import { requireAuth } from "../middleware/auth.middleware";
import { requireRole } from "../middleware/role.middleware";
import { resolveCompanyId } from "../middleware/team.middleware";
import {
  uploadJobDocument,
  getJobDocuments,
  deleteJobDocument,
} from "../controllers/job-document.controller";

const router = Router();

// Max 10MB per file, store in memory buffer
const upload = multer({
  storage: multer.memoryStorage(),
  limits: { fileSize: 10 * 1024 * 1024 },
});

router.use(requireAuth);

// Employer — upload document to their job
router.post(
  "/job/:jobId/documents",
  requireRole("employer"),
  resolveCompanyId,
  upload.single("file"),
  uploadJobDocument
);

// Employer — delete a document from their job
router.delete(
  "/job/:jobId/documents/:docId",
  requireRole("employer"),
  resolveCompanyId,
  deleteJobDocument
);

// Both employer and hired freelancer — get documents for a job
router.get("/job/:jobId/documents", getJobDocuments);

export default router;

import { useState, useRef } from "react";
import { useQuery, useMutation, useQueryClient } from
  "@tanstack/react-query";

// Inside the job card component:
const queryClient = useQueryClient();
const fileInputRef = useRef<HTMLInputElement>(null);
const [showUploadWarning, setShowUploadWarning] = useState(false);
const [pendingFile, setPendingFile] = useState<File | null>(null);
const [docType, setDocType] = useState("function_sheet");
const [uploading, setUploading] = useState(false);

// Fetch documents for this job
const { data: documents = [] } = useQuery<JobDocument[]>({
  queryKey: [`/api/job/${job.id}/documents`],
});

const handleFileSelect = (
  e: React.ChangeEvent<HTMLInputElement>
) => {
  const file = e.target.files?.[0];
  if (!file) return;
  setPendingFile(file);
  setShowUploadWarning(true); // show warning BEFORE uploading
  // Reset input so same file can be re-selected if needed
  e.target.value = "";
};

const confirmUpload = async () => {
  if (!pendingFile) return;
  setShowUploadWarning(false);
  setUploading(true);
  try {
    const formData = new FormData();
    formData.append("file", pendingFile);
    formData.append("documentType", docType);
    await fetch(`/api/job/${job.id}/documents`, {
      method: "POST",
      body: formData,
    });
    queryClient.invalidateQueries({
      queryKey: [`/api/job/${job.id}/documents`],
    });
  } finally {
    setUploading(false);
    setPendingFile(null);
  }
};

const deleteDocument = async (docId: number) => {
  await fetch(`/api/job/${job.id}/documents/${docId}`, {
    method: "DELETE",
  });
  queryClient.invalidateQueries({
    queryKey: [`/api/job/${job.id}/documents`],
  });
};
