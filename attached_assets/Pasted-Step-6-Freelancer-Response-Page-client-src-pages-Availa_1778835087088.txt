6.1 — New public API endpoint for token lookup
Add this endpoint to enquiry.route.ts (before the POST /respond/:token route):
// Add to enquiry.route.ts:
enquiryRouter.get('/respond/:token', getResponseByToken); // PUBLIC — no auth

// Add to enquiry.controller.ts:
export const getResponseByTokenHandler = async (req: Request, res: Response) => {
  try {
    const { token } = req.params;
    const row = await storage.getResponseByToken(token);
    if (!row) return res.status(404).json({ error: 'Token not found' });
    // Do not expose the token itself in the response
    const { response: resp, enquiry, user } = row;
    return res.json({
      response: {
        id: resp.id,
        response: resp.response,
        respondedAt: resp.respondedAt,
        responseNote: resp.responseNote,
      },
      enquiry: {
        id: enquiry.id,
        eventTitle: enquiry.eventTitle,
        eventDate: enquiry.eventDate,
        eventEndDate: enquiry.eventEndDate,
        callTime: enquiry.callTime,
        venueAddress: enquiry.venueAddress,
        roleRequired: enquiry.roleRequired,
        agreedRate: enquiry.agreedRate,
        additionalNotes: enquiry.additionalNotes,
        status: enquiry.status,
      },
      freelancer: { firstName: user.first_name },
    });
  } catch (error) {
    console.error('getResponseByToken error:', error);
    return res.status(500).json({ error: 'Internal server error' });
  }
};
