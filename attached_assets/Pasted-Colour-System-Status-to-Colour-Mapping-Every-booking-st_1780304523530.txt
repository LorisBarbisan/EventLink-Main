Colour System — Status to Colour Mapping
Every booking status maps to a specific colour. These colours must be used consistently everywhere — calendar events, popup badge, and booking card badge.

Colour	Status	Meaning
	enquired	Availability check sent, not yet confirmed — amber/orange
	confirmed	Booking confirmed — dark green
	briefed	Brief sent to freelancer — EventLink navy
	completed	Event complete — neutral grey
	cancelled	Booking cancelled — red

⚠️  These exact hex values must be used: enquired #FFA500, confirmed #1A6B3C, briefed #1E3A5F, completed #6B7280, cancelled #EF4444. Define them as a constant object STATUS_COLOURS at the top of the calendar component file.
 
Step 1 — API Endpoint (server/api/routes/booking.route.ts)
Add one new endpoint to the existing booking route. This returns all bookings for the logged-in employer, with freelancer name and profile photo included, formatted for the calendar.

⚠️  Check the existing booking.route.ts and booking.controller.ts before adding. If a 'get all bookings for employer' endpoint already exists, check whether it includes freelancer name and photo. If it does, use it and skip this step. If not, add the endpoint below.

1.1 — Storage method (server/storage.ts)
Add this method in the FMS bookings section:
async getBookingsForCalendar(employerId: number) {
  const results = await db
    .select({
      booking: bookings,
      freelancerFirstName: users.first_name,
      freelancerLastName: users.last_name,
      freelancerPhoto: freelancer_profiles.profile_photo_url,
      freelancerTitle: freelancer_profiles.title,
    })
    .from(bookings)
    .leftJoin(users, eq(bookings.freelancerId, users.id))
    .leftJoin(freelancer_profiles, eq(freelancer_profiles.user_id, users.id))
    .where(eq(bookings.employerId, employerId))
    .orderBy(bookings.createdAt);
  return results.map(r => ({
    ...r.booking,
    freelancerName: `${r.freelancerFirstName ?? ''} ${r.freelancerLastName ?? ''}`.trim(),
    freelancerPhoto: r.freelancerPhoto ?? null,
    freelancerTitle: r.freelancerTitle ?? null,
  }));
},

1.2 — Controller function
Add to server/api/controllers/booking.controller.ts:
export const getBookingsForCalendar = async (req: Request, res: Response) => {
  try {
    const employerId = req.user?.id;
    if (!employerId) return res.status(401).json({ error: 'Unauthorised' });
    const bookings = await storage.getBookingsForCalendar(employerId);
    return res.json(bookings);
  } catch (error: any) {
    console.error('getBookingsForCalendar error:', error.message);
    return res.status(500).json({ error: 'Internal server error' });
  }
};

const [selectedBooking, setSelectedBooking] = useState<any>(null);
const [popupOpen, setPopupOpen] = useState(false);

const { data: bookings = [], isLoading } = useQuery({
  queryKey: ['/api/bookings/calendar'],
  queryFn: () => apiRequest('/api/bookings/calendar'),
});

// Transform bookings to react-big-calendar event format
const events = bookings.map((b: any) => ({
  id: b.id,
  title: b.freelancerName || 'Unnamed',
  // Use eventDate as both start and end (single day events)
  // Parse date string carefully — bookings store dates as text (YYYY-MM-DD)
  start: new Date(b.eventDate ?? b.createdAt),
  end: new Date(b.eventDate ?? b.createdAt),
  allDay: true,
  resource: b, // store full booking data for popup
  status: b.status,
  colour: STATUS_COLOURS[b.status] ?? '#6B7280',
}));

// Event style getter — applies status colour to each event
const eventStyleGetter = (event: any) => ({

    fontSize: '12px',

// Handle event click — open popup

  setSelectedBooking(event.resource);
  setPopupOpen(true);

// Render
return (
  <div className='h-[700px] p-4'>
    {/* Colour legend */}
    <div className='flex flex-wrap gap-3 mb-4'>
      {Object.entries(STATUS_COLOURS).map(([status, colour]) => (
        <div key={status} className='flex items-center gap-1.5'>
          <div className='h-3 w-3 rounded-full' style={{ backgroundColor: colour }} />
          <span className='text-xs capitalize text-muted-foreground'>{status}</span>
        </div>
      ))}
    </div>
    {isLoading ? (
      <div className='flex items-center justify-center h-full'>
        <Loader2 className='h-6 w-6 animate-spin text-muted-foreground' />
      </div>
    ) : (
      <Calendar
        localizer={localizer}
        events={events}
        defaultView={Views.MONTH}
        views={[Views.MONTH, Views.WEEK, Views.AGENDA]}
        style={{ height: '100%' }}
        eventPropGetter={eventStyleGetter}
        onSelectEvent={handleSelectEvent}
        culture='en-GB'
        formats={{
          monthHeaderFormat: (date: Date) => format(date, 'MMMM yyyy'),
          dayHeaderFormat: (date: Date) => format(date, 'EEE d MMM'),
          agendaDateFormat: (date: Date) => format(date, 'EEE d MMM yyyy'),
        }}
        popup // show overflow popup when too many events on one day
        tooltipAccessor={(event: any) => `${event.title} — ${event.status}`}
      />
    )}
    {selectedBooking && (
      <BookingEventPopup
        booking={selectedBooking}
        open={popupOpen}
        onOpenChange={(open) => { setPopupOpen(open); if (!open) setSelectedBooking(null); }}
      />
    )}
  </div>
);
